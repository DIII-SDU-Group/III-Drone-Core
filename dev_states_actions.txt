Actions:

Takeoff:
Preconditions:
	* target_altitude >= minimum_target_altitude
	* state == on_ground_non_offboard - done
Cancel conditions:
	* false
Success:
	* armed and offboard and reached position
Fail:
	* Timeout
Nested actions:
	* arm
	* set offboard
	* reach position

Landing:
Preconditions:
	* state == hovering || state = hovering_under_cable
Cancel conditions:
	* false
Success:
	* !armed and !offboard	
Fail:
	* Timeout
	
FlyToPosition:
Preconditions:
	* state == hovering || state == hovering_under_cable
Cancel conditions:
	* true
Success:
	* Reached position
	
	
FlyUnderCable:
Preconditions:
	* state == hovering || state == hovering_under_cable
	* Cable is visible
Cancel conditions:
	* true
Success:
	* Reached position
Fail:
	* Cable not visible
Fallback:
	* FlyUnderCable
	Fallback:
		* FlyToPosition z: -1
	
	
CableLanding:
Preconditions:
	* state == hovering_under_cable
	* cable_id == current target_cable_id
	* Cable is visible
Cancel conditions:
	* True
Success: 
	* Reached position or (use gripper conditions && gripper is closed)
Fail:
	* Violate safety margins or cable not visible
Fallback:
	* If cable is visible:
		FlyUnderCable
		Fallback:
			* FlyToPosition z: -1
	* If cable is not visible:
		FlyToPosition z: -1
Nested actions:
	* Close gripper if use gripper conditions
	* Reach cable
		
	
CableTakeoff:
Preconditions:
	* armed and offboard
	* State == on_cable_armed
	* Gripper is open if use gripper conditions
	* Target cable distance > 1
Cancel:
	* false
Success:
	* Reached position and cable is visible
Fail:
	* Cable is not visible
Fallback:
	* FlyToPosition z: -1
	

DisarmOnCable:
Preconditions:
	* state == on_cable_armed
	* use gripper conditions
	* Gripper is closed
	* armed and offboard
Cancel:
	* false
Success:
	* Gripper is closed
	* !armed
Fail:
	* Timeout
Nested actions:
	* land


ArmOnCable:
Preconditions:
	* state == on_cable_disarmed
	* !armed
	* On cable
Cancel:
	* false
Success:
	* On cable
	* Armed
	* Offboard
Fail:
	* Timeout and (!armed or !offboard)
Nested actions:
	* arm
	* set offboard
	
	
States:
init:

on_ground_non_offboard:
	* !offboard and altitude < landed_altitude_threshold
	
in_flight_non_offboard:
	* !offboard and altitude > landed_altitude_threshold
	
hovering:
	* offboard and altitude > landed_altitude_threshold
	
arming:
	* Arming called but has not succeeded
	
on_cable_disarmed:
	* !armed and altitude >landed_altitude_threshold
		* (pl_pos - gripper_pos).norm() < landed_on_powerline_non_offboard_max_euc_distance
		* and gripper is closed
		* or disarm on cable succeeded
		
setting_offboard:
	* armed and offboard called but has not succeeded
	
taking_off:
	* armed and offboard and takeoff called but has not succeeded yet
	
in_positional_flight:
	* armed and offboard and during execution of FlyToPosition or FlyUnderCable
	
landing:
	* land called and altitude > landed_altitude_threshold
	
hovering_under_cable:
	* armed and offboard and hovering_under_cable and cable visible
	
during_cable_landing:
	* armed and offboard and cable visible and cable landing action is executing
	
during_cable_takeoff:
	* armed and offboard and cable visible and cable takeoff action is executing
	
on_cable_armed:
	* armed and offboard and altitude > landed_altitude_threshold and 
		((pl_pos - gripper_pos).norm() < landed_on_powerline max euc distance) or cable landing action succeeded
		
during_cable_takeoff:
	* armed and offboard and cable is visible and cable takeoff action is executing
	
disarming_on_cable:
	* on cable and armed and disarm on cable called
	
arming_on_cable:
	* on cable and !armed and arm called
	
setting_offboard_on_cable
	* on cable and armed and !offboard and set offboard called
	
	
	
Conditions:
* armed - done
* offboard - done
* on ground - done
* has target cable
* cable is visible
* is on cable

* is arming
* is setting offboard
* is taking off
* is executing FlyToPosition
* is executing FlyUnderCable
* is executing CableLanding
* is executing CableTakeoff
* is landing
* is idle
