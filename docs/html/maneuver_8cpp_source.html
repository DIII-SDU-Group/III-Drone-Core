192void Maneuver::Start() {
193
194 terminated_ = false;
195 success_ = false;
196
197 switch(maneuver_type_) {
198 case MANEUVER_TYPE_FLY_TO_POSITION: {
199 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::FlyToPosition>>(goal_handle_);
200 if (!goal_handle->is_executing()) goal_handle->execute();
201 break;
202 }
203
204 case MANEUVER_TYPE_FLY_TO_OBJECT: {
205 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::FlyToObject>>(goal_handle_);
206 if (!goal_handle->is_executing()) goal_handle->execute();
207 break;
208 }
209
210 case MANEUVER_TYPE_CABLE_LANDING: {
211 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::CableLanding>>(goal_handle_);
212 if (!goal_handle->is_executing()) goal_handle->execute();
213 break;
214 }
215
216 case MANEUVER_TYPE_CABLE_TAKEOFF: {
217 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::CableTakeoff>>(goal_handle_);
218 if (!goal_handle->is_executing()) goal_handle->execute();
219 break;
220 }
221
222 case MANEUVER_TYPE_HOVER: {
223 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::Hover>>(goal_handle_);
224 if (!goal_handle->is_executing()) goal_handle->execute();
225 break;
226 }
227
228 case MANEUVER_TYPE_HOVER_BY_OBJECT: {
229 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::HoverByObject>>(goal_handle_);
230 if (!goal_handle->is_executing()) goal_handle->execute();
231 break;
232 }
233
234 case MANEUVER_TYPE_HOVER_ON_CABLE: {
235 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<iii_drone_interfaces::action::HoverOnCable>>(goal_handle_);
236 if (!goal_handle->is_executing()) goal_handle->execute();
237 break;
238 }
239
240 default:
241 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"),"Maneuver::Start(): Unknown maneuver type");
242 break;
243
244 }
245
246 start_time_ = rclcpp::Clock().now();
247
248}
