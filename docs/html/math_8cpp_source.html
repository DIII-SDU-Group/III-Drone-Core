16
17 float cos_yaw = cos(eul[2]);

19 float cos_roll = cos(eul[0]);
20 float sin_yaw = sin(eul[2]);
21 float sin_pitch = sin(eul[1]);
22 float sin_roll = sin(eul[0]);
23
24 rotation_matrix_t mat;
25
26 mat(0,0) = cos_pitch*cos_yaw;
27 mat(0,1) = sin_roll*sin_pitch*cos_yaw-cos_roll*sin_yaw;
28 mat(0,2) = cos_roll*sin_pitch*cos_yaw+sin_roll*sin_yaw;
29 mat(1,0) = cos_pitch*sin_yaw;
30 mat(1,1) = sin_roll*sin_pitch*sin_yaw+cos_roll*cos_yaw;
31 mat(1,2) = cos_roll*sin_pitch*sin_yaw-sin_roll*cos_pitch; // wrong? cos_roll*sin_pitch*sin_yaw-sin_roll*cos_yaw
32 mat(2,0) = -sin_pitch;
33 mat(2,1) = sin_roll*cos_pitch;
34 mat(2,2) = cos_roll*cos_pitch;
35

108const quaternion_t iii_drone::math::matToQuat(const rotation_matrix_t & R) {
109
110 float tr = R(0,0) + R(1,1) + R(2,2);
111
112 float qw, qx, qy, qz;
113
114 if (tr > 0) {
115
116 float S = sqrt(tr+1.0) * 2; // S=4*qw
117 qw = 0.25 * S;
118 qx = (R(2,1) - R(1,2)) / S;
119 qy = (R(0,2) - R(2,0)) / S;
120 qz = (R(1,0) - R(0,1)) / S;
121
122 } else if ((R(0,0) > R(1,1)) && (R(0,0) > R(2,2))) {
123
124 float S = sqrt(1.0 + R(0,0) - R(1,1) - R(2,2)) * 2; // S=4*qx
125 qw = (R(2,1) - R(1,2)) / S;
126 qx = 0.25 * S;
127 qy = (R(0,1) + R(1,0)) / S;
128 qz = (R(0,2) + R(2,0)) / S;
129
130 } else if (R(1,1) > R(2,2)) {
131
132 float S = sqrt(1.0 + R(1,1) - R(0,0) - R(2,2)) * 2; // S=4*qy
133 qw = (R(0,2) - R(2,0)) / S;
134 qx = (R(0,1) + R(1,0)) / S;
135 qy = 0.25 * S;
136 qz = (R(1,2) + R(2,1)) / S;
137
138 } else {
139
140 float S = sqrt(1.0 + R(2,2) - R(0,0) - R(1,1)) * 2; // S=4*qz
141 qw = (R(1,0) - R(0,1)) / S;
142 qx = (R(0,2) + R(2,0)) / S;
143 qy = (R(1,2) + R(2,1)) / S;
144 qz = 0.25 * S;
145
146 }
147
148 quaternion_t quat(qw, qx, qy, qz);
149
150 return quat;
151
152}

154const quaternion_t iii_drone::math::eulToQuat(const euler_angles_t & eul) {
155
156 // Abbreviations for the various angular functions
157 float cy = cos(eul(2) * 0.5);
158 float sy = sin(eul(2) * 0.5);
159 float cp = cos(eul(1) * 0.5);
160 float sp = sin(eul(1) * 0.5);
161 float cr = cos(eul(0) * 0.5);
162 float sr = sin(eul(0) * 0.5);
163
164 quaternion_t q;
165 q(0) = cr * cp * cy + sr * sp * sy;
166 q(1) = sr * cp * cy - cr * sp * sy;
167 q(2) = cr * sp * cy + sr * cp * sy;
168 q(3) = cr * cp * sy - sr * sp * cy;
169
170 return q;
171
172}
