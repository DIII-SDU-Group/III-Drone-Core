42) const {
43

46 }
47
48 hover_on_cable_maneuver_params_t params(maneuver.maneuver_params());

51 return false;

53

176void HoverOnCableManeuverServer::publishResultAndFinalize(
177 Maneuver &maneuver,
178 maneuver_result_type_t maneuver_result_type
179) {
180
181 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<HoverOnCable>>(maneuver.goal_handle());
182 auto result = std::make_shared<HoverOnCable::Result>();
183
184 switch (maneuver_result_type){
185
186 case MANEUVER_RESULT_TYPE_SUCCEED:
187 goal_handle->succeed(result);
188 break;
189 case MANEUVER_RESULT_TYPE_ABORT:
190 goal_handle->abort(result);
191 awareness_handler()->ClearTarget();
192 break;
193 case MANEUVER_RESULT_TYPE_CANCEL:
194 goal_handle->canceled(result);
195 awareness_handler()->ClearTarget();
196 break;
197 }
198
199}
