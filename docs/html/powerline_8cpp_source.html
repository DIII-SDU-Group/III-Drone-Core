267void Powerline::ComputeInterLinePositions() {
268
269 quaternion_t direction = pl_dir_history_[0];
270
271 quaternion_t q_pl_to_drone = quatInv(direction);
272 rotation_matrix_t R_pl_to_drone = quatToMat(q_pl_to_drone);
273
274 std::unique_lock<std::shared_mutex> lines_lock(lines_mutex_);
275
276 if (lines_.empty()) {
277
278 return;
279
280 }
281
282 for (unsigned int i = 0; i < lines_.size()-1; i++) {
283
284 bool in_fov = lines_[i].IsInFOV();
285
286 if (!in_fov) {
287
288 continue;
289
290 }
291
292 for (unsigned int j = i+1; j < lines_.size(); j++) {
293
294 if (!lines_[j].IsInFOV()) {
295
296 continue;
297
298 }
299
300 bool ilp_found = false;
301
302 for (unsigned int k = 0; k < inter_line_positions_.size(); k++) {
303
304 if (inter_line_positions_[k].line_id_1 == lines_[i].id() && inter_line_positions_[k].line_id_2 == lines_[j].id()) {
305
306 vector_t vec = lines_[j].position() - lines_[i].position();
307 vec = R_pl_to_drone * vec;
308
309 inter_line_positions_[k].inter_line_position_window.push_back(vec);
310
311 while(inter_line_positions_[k].inter_line_position_window.size() > (unsigned int)parameters_->inter_pos_window_size()) {
312
313 inter_line_positions_[k].inter_line_position_window.erase(inter_line_positions_[k].inter_line_position_window.begin());
314
315 }
316
317 ilp_found = true;
318 break;
319
320 } else if (inter_line_positions_[k].line_id_1 == lines_[j].id() && inter_line_positions_[k].line_id_2 == lines_[i].id()) {
321
322 vector_t vec = lines_[i].position() - lines_[j].position();
323 vec = R_pl_to_drone * vec;
324
325 inter_line_positions_[k].inter_line_position_window.push_back(vec);
326
327 while(inter_line_positions_[k].inter_line_position_window.size() > (unsigned int)parameters_->inter_pos_window_size()) {
328
329 inter_line_positions_[k].inter_line_position_window.erase(inter_line_positions_[k].inter_line_position_window.begin());
330
331 }
332
333 ilp_found = true;
334 break;
335
336 }
337
338 }
339
340 if (!ilp_found) {
341
342 throw std::runtime_error("Inter line position not found.");
343
344 }
345 }
346 }

508void Powerline::predictLines() {
509
510 vector_t delta_position;
511 quaternion_t delta_quat, q_drone_to_pl;
512
513 pose_t drone_pose = drone_pose_history_[0];
514 pose_t last_drone_pose = drone_pose_history_[-1];
515
516 rotation_matrix_t W_R_D1 = quatToMat(last_drone_pose.orientation);
517 rotation_matrix_t W_R_D2 = quatToMat(drone_pose.orientation);
518 rotation_matrix_t D2_R_W = W_R_D2.transpose();
519
520 delta_quat = matToQuat(D2_R_W*W_R_D1);
521
522 delta_position = drone_pose.position - last_drone_pose.position;
523 delta_position = D2_R_W * delta_position;
524
525 q_drone_to_pl = pl_dir_history_[0];
526
527 rotation_matrix_t R_drone_to_pl = quatToMat(q_drone_to_pl);
528
529 plane_t projection_plane = projection_plane_;
530
531 {
532
533 std::unique_lock<std::shared_mutex> lock(lines_mutex_);
534
535 std::vector<unsigned int> non_visible_line_indices;
536
537 for (unsigned int i = 0; i < lines_.size(); i++) {
538
539 if (lines_[i].IsInFOV()) {
540
541 lines_[i].Predict(delta_position, delta_quat, projection_plane);
542
543 } else {
544
545 non_visible_line_indices.push_back(i);
546
547 }
548 }
549
550 for (unsigned int i = 0; i < non_visible_line_indices.size(); i++) {
551
552 unsigned int idx = non_visible_line_indices[i];
553
554 std::vector<point_t> expected_positions;
555
556 for (unsigned int j = 0; j < inter_line_positions_.size(); j++) {
557
558 if (inter_line_positions_[j].inter_line_position_window.size() < 1)
559 continue;
560
561 int id1 = inter_line_positions_[j].line_id_1;
562 int id2 = inter_line_positions_[j].line_id_2;
563
564 bool id1_match = id1 == lines_[idx].id();
565 bool id2_match = id2 == lines_[idx].id();
566
567 if (id1_match || id2_match) {
568
569 int ref_line_id = id1_match ? id2 : id1;
570
571 point_t reference_line_point;
572 bool ref_line_point_found = false;
573
574 for (unsigned int k = 0; k < lines_.size(); k++) {
575
576 if (k == idx)
577 continue;
578
579 if (lines_[k].id() == ref_line_id && lines_[k].IsInFOV()) {
580
581 reference_line_point = lines_[k].position();
582 ref_line_point_found = true;
583
584 break;
585
586 }
587 }
588
589 if (ref_line_point_found) {
590
591 vector_t mean_vec = inter_line_positions_[j].inter_line_position_window[0];
592
593 for (unsigned int k = 1; k < inter_line_positions_[j].inter_line_position_window.size(); k++) {
594
595 mean_vec += inter_line_positions_[j].inter_line_position_window[k];
596
597 }
598
599 mean_vec /= inter_line_positions_[j].inter_line_position_window.size();
600
601 float mult = id1_match ? -1. : 1.;
602
603 mean_vec *= mult;
604
605 mean_vec = R_drone_to_pl * mean_vec;
606
607 point_t expected_pos = reference_line_point + mean_vec;
608
609 expected_positions.push_back(expected_pos);
610
611 }
612 }
613 }
614
615 if (expected_positions.size() > 0) {
616
617 point_t mean_pos = expected_positions[0];
618
619 for (unsigned int j = 1; j < expected_positions.size(); j++) {
620
621 mean_pos += expected_positions[j];
622
623 }
624
625 mean_pos /= expected_positions.size();
626
627 lines_[idx].SetPosition(mean_pos);
628
629 } else {
630
631 lines_[idx].Predict(delta_position, delta_quat, projection_plane);
632
633 }
634 }
635 }
