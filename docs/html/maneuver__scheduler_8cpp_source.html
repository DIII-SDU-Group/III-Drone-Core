530void ManeuverScheduler::progressScheduler() {
531
532 static bool waiting_for_maneuver_to_start = false;
533 static std::string waiting_for_maneuver_to_start_action_name;
534
535 std::unique_lock<std::shared_mutex> lck(maneuver_mutex_);
536
537 auto on_no_maneuver = [this](Maneuver previous_maneuver) {
538
539 if (!previous_maneuver.success()) {
540
541 no_maneuver_idle_cnt_ = parameters_->no_maneuver_idle_cnt_s() * 1000 / parameters_->maneuver_execution_period_ms();
542
543 } else {
544
545 switch (previous_maneuver.maneuver_type()) {
546
547 case MANEUVER_TYPE_HOVER: {
548 hover_maneuver_params_t maneuver_params(previous_maneuver.maneuver_params());
549
550 no_maneuver_idle_cnt_ = maneuver_params.duration_s * 1000 / parameters_->maneuver_execution_period_ms();
551
552 break;
553 }
554 case MANEUVER_TYPE_HOVER_BY_OBJECT: {
555 hover_by_object_maneuver_params_t maneuver_params(previous_maneuver.maneuver_params());
556
557 no_maneuver_idle_cnt_ = maneuver_params.duration_s * 1000 / parameters_->maneuver_execution_period_ms();
558
559 break;
560 }
561 case MANEUVER_TYPE_HOVER_ON_CABLE: {
562 hover_on_cable_maneuver_params_t maneuver_params(previous_maneuver.maneuver_params());
563
564 no_maneuver_idle_cnt_ = maneuver_params.duration_s * 1000 / parameters_->maneuver_execution_period_ms();
565
566 break;
567 }
568 case MANEUVER_TYPE_NONE:
569 if (*no_maneuver_idle_cnt_ >= 0)
570 (*no_maneuver_idle_cnt_)--;
571 break;
572
573 default:
574 no_maneuver_idle_cnt_ = parameters_->no_maneuver_idle_cnt_s() * 1000 / parameters_->maneuver_execution_period_ms();
575 break;
576
577 }
578
579 }
580
581 if (*no_maneuver_idle_cnt_ < 0) {
582
583 maneuver_execution_timer_->cancel();
584
585 }
586 };
587
588 auto on_failure = [this, &on_no_maneuver](Maneuver previous_maneuver) {
589
590 previous_maneuver.Terminate(false);
591
592 cancelAllPendingManeuvers();
593
594 reference_callback_token_.DenyAll();
595
596 on_no_maneuver(previous_maneuver);
597
598 };
599
600 // Check if a maneuver is executing
601 if (maneuverIsExecutingOrPending()) {
602
603 // Check if the current maneuver is completed
604 if(current_maneuver_->terminated()) {
605
606 if (!reference_callback_token_.master_has_token()) {
607
608 std::string msg = "ManeuverScheduler::progressScheduler(): master does not have token.";
609
610 RCLCPP_FATAL(rclcpp::get_logger("rclcpp"), msg.c_str());
611
612 throw std::runtime_error(msg);
613
614 }
615
616 // Check if the maneuver was successful
617 if (current_maneuver_->success()) {
618
619 // Check if there are more maneuvers in the queue
620 Maneuver maneuver;
621 if (maneuver_queue_->Pop(maneuver)) {
622
623 // Progress to the next maneuver
624 current_maneuver_ = maneuver;
625
626 // Start the maneuver execution:
627 current_maneuver_->Start();
628
629 // Find the maneuver server:
630 auto registered_maneuver = registered_maneuvers_.find(maneuver.maneuver_type());
631
632 // Get the action name:
633 waiting_for_maneuver_to_start_action_name = registered_maneuver->second->action_name();
634
635 // Set waiting for start:
636 waiting_for_maneuver_to_start = true;
637
638 // Reset no_maneuver_idle_cnt:
639 *no_maneuver_idle_cnt_ = -1;
640
641
642 } else {
643
644 on_no_maneuver(current_maneuver_);
645
646 current_maneuver_ = Maneuver();
647
648 }
649
650 } else {
651
652 on_failure(current_maneuver_);
653
654 }
655
656 } else {
657 // Current maneuver has not terminated.
658
659 if (waiting_for_maneuver_to_start) {
660
661 // Check that the maneuver has a goal handle:
662 if (current_maneuver_->goal_handle() == nullptr) {
663
664 // Check that the update timeout has elapsed:
665 rclcpp::Time maneuver_creation_time = current_maneuver_->creation_time();
666 rclcpp::Time current_time = node_->now();
667
668 if ((current_time - maneuver_creation_time).seconds() > parameters_->maneuver_register_update_timeout_s()) {
669
670 on_failure(current_maneuver_);
671
672 }
673
674 return;
675
676 }
677
678 if (reference_callback_token_.has_requested_token(waiting_for_maneuver_to_start_action_name)) {
679
680 // Give the token:
681 reference_callback_token_.Give(waiting_for_maneuver_to_start_action_name);
682
683 // Reset the flag:
684 waiting_for_maneuver_to_start = false;
685
686 } else {
687
688 rclcpp::Time maneuver_start_time = current_maneuver_->start_time();
689 rclcpp::Time current_time = node_->now();
690
691 if ((current_time - maneuver_start_time).seconds() > parameters_->maneuver_start_timeout_s()) {
692
693 on_failure(current_maneuver_);
694
695 }
696 }
697 }
698 }
699
700 } else {
701
702 if (current_maneuver_->maneuver_type() != MANEUVER_TYPE_NONE) {
703
704 std::string msg = "ManeuverScheduler::progressScheduler(): maneuver is not executing or pending, but current maneuver is not MANEUVER_TYPE_NONE.";
705
706 RCLCPP_FATAL(rclcpp::get_logger("rclcpp"), msg.c_str());
707
708 throw std::runtime_error(msg);
709
710 }
711
712 on_no_maneuver(current_maneuver_);
713
714 }
