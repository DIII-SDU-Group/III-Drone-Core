40bool HoverByObjectManeuverServer::CanExecuteManeuver(
41 const Maneuver & maneuver,
42 const combined_drone_awareness_t & drone_awareness
43) const {
44
45 if (maneuver.maneuver_type() != MANEUVER_TYPE_HOVER_BY_OBJECT) {
46 return false;
47 }
48
49 hover_by_object_maneuver_params_t params(maneuver.maneuver_params());
50
51 if (params.target_adapter.target_type() != TARGET_TYPE_CABLE) {
52 return false;
53 }
54
55 if (!drone_awareness.offboard) {
56 return false;
57 }
58
59 if (!drone_awareness.armed) {
60 return false;
61 }
62
63 if (!drone_awareness.in_flight()) {
64 return false;
65 }
66
67 if (!drone_awareness.has_target()) {
68 return false;
69 }
70
71 if (drone_awareness.target_adapter.target_type() != TARGET_TYPE_CABLE) {
72 return false;
73 }
74
75 transform_matrix_t target_transform;
76
77 try {
78
79 target_transform = awareness_handler()->ComputeTargetTransform(target_adapter_);
80
81 } catch (const std::runtime_error &e) {
82
83 return false;
84
85 }
86
87 if (!validateTargetTransform(
88 target_transform,
89 drone_awareness.state
90 )) {
91
92 return false;
93
94 }
95
96 return true;

155iii_drone::control::Reference HoverByObjectManeuverServer::GetReference(const iii_drone::control::State & state) {
156
157 if (!has_on_fail_callback_) {
158
159 throw std::runtime_error("No on fail callback registered for HoverByObjectManeuverServer");
160
161 }
162
163 if (!has_target_) {
164
165 throw std::runtime_error("No target set for HoverByObjectManeuverServer");
166
167 }
168
169 transform_matrix_t target_transform;
170
171 try {
172
173 target_transform = awareness_handler()->ComputeTargetTransform(target_adapter_);
174
175 } catch (const std::runtime_error &e) {
176
177 has_target_ = false;
178
179 }
180
181 if (!validateTargetTransform(
182 target_transform,
183 awareness_handler()->GetState()
184 )) {
185
186 has_target_ = false;
187
188 }
189
190 vector_t position;
191 double yaw;
192 vector_t velocity;
193 double yaw_rate;
194 vector_t acceleration;
195 double yaw_acceleration;
196
197 if (has_target_) {
198
199 pose_t pose = poseFromTransformMatrix(target_transform);
200
201 position = pose.position;
202 yaw = quatToEul(pose.orientation)(2);
203
204 } else {
205
206 on_fail_callback_();
207
208 position = state.position();
209 yaw = state.yaw();
210
211 }
212
213 if (use_nans_) {
214
215 velocity = vector_t::Constant(NAN);
216 yaw_rate = NAN;
217 acceleration = vector_t::Constant(NAN);
218 yaw_acceleration = NAN;
219
220 } else {
221
222 velocity = vector_t::Zero();
223 yaw_rate = 0;
224 acceleration = vector_t::Zero();
225 yaw_acceleration = 0;
226
227 }
228
229 return iii_drone::control::Reference(
230 position,
231 yaw,
232 velocity,
233 yaw_rate,
234 acceleration,
235 yaw_acceleration
236 );

306void HoverByObjectManeuverServer::publishResultAndFinalize(
307 Maneuver & maneuver,
308 maneuver_result_type_t maneuver_result_type
309) {
310
311 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<HoverByObject>>(maneuver.goal_handle());
312 auto result = std::make_shared<HoverByObject::Result>();
313
314 switch (maneuver_result_type){
315
316 case MANEUVER_RESULT_TYPE_SUCCEED:
317 goal_handle->succeed(result);
318 break;
319 case MANEUVER_RESULT_TYPE_ABORT:
320 goal_handle->abort(result);
321 awareness_handler()->ClearTarget();
322 break;
323 case MANEUVER_RESULT_TYPE_CANCEL:
324 goal_handle->canceled(result);
325 awareness_handler()->ClearTarget();
326 break;
327 }
328
329}
