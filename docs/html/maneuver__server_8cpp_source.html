81template <typename ActionT>
82rclcpp_action::GoalResponse ManeuverServer::handleGoal(
83 const rclcpp_action::GoalUUID & uuid,
84 std::shared_ptr<const typename ActionT::Goal> goal
85) {
86
87 if (!running_) {
88 return rclcpp_action::GoalResponse::REJECT;
89 }
90
91 (void)uuid;
92
93 Maneuver maneuver(
94 maneuver_type(),
95 uuid
96 );
97
98 maneuver.SetFromGoal<ActionT>(goal);
99
100 bool executing_instantly;
101 bool success = register_maneuver_(maneuver, executing_instantly);
102
103 if (!success) {
104 return rclcpp_action::GoalResponse::REJECT;
105 }
106
107 if (executing_instantly) {
108 return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
109 }
110
111 return rclcpp_action::GoalResponse::ACCEPT_AND_DEFER;
112
113}

130template <typename ActionT>
131void ManeuverServer::handleAccepted(
132 const std::shared_ptr<rclcpp_action::ServerGoalHandle<ActionT>> goal_handle
133) {
134
135 Maneuver maneuver = Maneuver::FromGoalHandle<ActionT>(goal_handle);
136
137 if (!update_maneuver_(maneuver)) {
138 goal_handle->abort(std::make_shared<typename ActionT::Result>());
139 return;
140 }
141
142 std::thread{std::bind(
143 &ManeuverServer::asyncExecute<ActionT>,
144 this,
145 std::placeholders::_1
146 ),
147 goal_handle
148 }.detach();

152template <typename ActionT>
153void ManeuverServer::asyncExecute(
154 const std::shared_ptr<rclcpp_action::ServerGoalHandle<ActionT>> goal_handle
155) {
156
157 Maneuver maneuver = Maneuver::FromGoalHandle<ActionT>(goal_handle);
158
159 rclcpp::Rate rate = rclcpp::Rate(std::chrono::milliseconds(wait_for_execute_poll_ms_));
160
161 while(!goal_handle->is_executing()) {
162 rate.sleep();
163 if (!verify_maneuver_in_queue_(maneuver)) {
164 goal_handle->canceled(std::make_shared<typename ActionT::Result>());
165 maneuver.Terminate(false);
166 cancel_maneuver_(maneuver);
167 current_maneuver_ = Maneuver();
168
169 return;
170 }
171 }
172
173 std::unique_lock<std::mutex> lock(mutex_, std::try_to_lock);
174
175 if (!lock.owns_lock()) {
176 goal_handle->canceled(std::make_shared<typename ActionT::Result>());
177 return;
178 }
179
180 current_maneuver_ = maneuver;
181
182 if (!reference_callback_token_->Acquire()) {
183 goal_handle->canceled(std::make_shared<typename ActionT::Result>());
184 maneuver.Terminate(false);
185 cancel_maneuver_(maneuver);
186 current_maneuver_ = Maneuver();
187
188 return;
189 }
190
191 startExecution(maneuver);
192
193 Atomic<std::function<Reference(const State &)>> &get_reference_callback = reference_callback_token_->resource();
194
195 get_reference_callback.Store(
196 std::bind(
197 &ManeuverServer::computeReference,
198 this,
199 std::placeholders::_1
200 )
201 );
202
203 bool success;
204
205 while(true) {
206
207 if (!verify_maneuver_active_(maneuver)) {
208 publishResultAndFinalize(
209 maneuver,
210 MANEUVER_RESULT_TYPE_CANCEL
211 );
212 maneuver.Terminate(false);
213 cancel_maneuver_(maneuver);
214 current_maneuver_ = Maneuver();
215 return;
216 }
217
218 if (goal_handle->is_canceling()) {
219 success = false;
220 publishResultAndFinalize(
221 maneuver,
222 MANEUVER_RESULT_TYPE_CANCEL
223 );
224 break;
225 }
226
227 if (hasSucceeded(maneuver)) {
228 success = true;
229 publishResultAndFinalize(
230 maneuver,
231 MANEUVER_RESULT_TYPE_SUCCEED
232 );
233 break;
234 }
235
236 if (hasFailed(maneuver)) {
237 success = false;
238 publishResultAndFinalize(
239 maneuver,
240 MANEUVER_RESULT_TYPE_ABORT
241 );
242 break;
243 }
244
245 publishFeedback(maneuver);
246
247 rate.sleep();
248
249 }
250
251 if (success) {
252 registerReferenceCallbackOnSuccess(maneuver);
253 }
254
255 reference_callback_token_->Release();
256
257 maneuver.Terminate(success);
258
259 done_callback_(maneuver);
260
261 current_maneuver_ = Maneuver();
262
263}

270void ManeuverServer::createServer() {
271
272 // Create action server
273 std::shared_ptr<rclcpp_action::Server<ActionT>> server = rclcpp_action::create_server<ActionT>(
274 node_,
275 action_name_,
276 std::bind(
277 &ManeuverServer::handleGoal<ActionT>,
278 this,
279 std::placeholders::_1,
280 std::placeholders::_2
281 ),
282 std::bind(
283 &ManeuverServer::handleCancel<ActionT>,
284 this,
285 std::placeholders::_1
286 ),
287 std::bind(
288 &ManeuverServer::handleAccepted<ActionT>,
289 this,
290 std::placeholders::_1
291 )
292 );
293
294 server_ = std::static_pointer_cast<void>(server);

296}
