62 const rclcpp_action::GoalUUID & uuid,

64) {

66 (void)uuid;

68 maneuver_t maneuver;

70 maneuver.set_from_goal<ActionT>(goal);
71
72 if (behavior_state_->CanExecute(maneuver)) {

117template <typename ActionT>
118void ManeuverServer<ActionT>::execute(
119 const std::shared_ptr<rclcpp_action::ServerGoalHandle<ActionT>> goal_handle
120) {
121
122 rclcpp::Rate rate = rclcpp::Rate(std::chrono::milliseconds(wait_for_execute_poll_ms_));
123
124 while(!goal_handle->is_executing()) {
125 rate.sleep();
126 }
127
128 reference_callback_token_.Acquire();
129
130 Atomic<std::function<Reference(const State &)>> &get_reference_callback = reference_callback_token_.resource();
131
132 get_reference_callback.Store(
133 std::bind(
134 &ManeuverServer::getReference,
135 this,
136 std::placeholders::_1
137 )
138 );
139
140 bool success;
141
142 auto feedback = std::make_shared<typename ActionT::Feedback>();
143 auto result = std::make_shared<typename ActionT::Result>();
144
145 while(true) {
146
147 if (goal_handle->is_canceling()) {
148 success = false;
149 getResult(
150 goal_handle,
151 success,
152 result
153 );
154 goal_handle->canceled(result);
155 break;
156 }
157
158 if (hasSucceeded(goal_handle)) {
159 success = true;
160 getResult(
161 goal_handle,
162 success,
163 result
164 );
165 goal_handle->succeed(result);
166 break;
167 }
168
169 if (hasFailed(goal_handle)) {
170 success = false;
171 getResult(
172 goal_handle,
173 success,
174 result
175 );
176 goal_handle->abort(result);
177 break;
178 }
179
180 getFeedback(
181 goal_handle,
182 feedback
183 );
184
185 goal_handle->publish_feedback(feedback);
186
187 rate.sleep();
188
189 }
190
191 done_callback_(success);
192
193}
