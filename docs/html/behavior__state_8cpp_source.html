39behavior_state_t BehaviorState::ProjectExpectedBehaviorState(maneuver_t maneuver) const {
40
41 std::vector<maneuver_t> maneuver_queue = (*maneuver_queue_).vector();
42
43 maneuver_t current_maneuver = current_maneuver_->Load();
44
45 if (maneuver_queue.size() == 0 || maneuver_queue[0] != current_maneuver) {
46
47 maneuver_queue.insert(
48 maneuver_queue.begin(),
49 current_maneuver
50 );
51
52 }
53
54 maneuver_queue.push_back(maneuver);
55
56 behavior_state_t bhst = behavior_state();
57 bhst.maneuver_is_executing = false;
58 bhst.current_maneuver = maneuver_t();
59
60 for (unsigned int i = 0; i < maneuver_queue.size(); i++) {
61
62 if (
63 !maneuverCanExecute(
64 maneuver_queue[i],
65 bhst
66 )
67 ) {
68
69 throw std::runtime_error("BehaviorState::ProjectExpectedBehaviorState: maneuver queue is invalid");
70
71 }
72
73 bhst = ProjectExpectedBehaviorState(
74 maneuver_queue[i],
75 bhst
76 );
77
78 }

90
91 bhst_out.maneuver_is_executing = false;

103 bhst_out.armed = false;
104 bhst_out.offboard = false;
105 bhst_out.drone_location = DRONE_LOCATION_ON_GROUND;

109 bhst_out.armed = true;

111 bhst_out.drone_location = DRONE_LOCATION_IN_FLIGHT;

115 bhst_out.armed = true;
116 bhst_out.offboard = true;

230bool BehaviorState::maneuverCanExecute(
231 iii_drone::control::maneuver::maneuver_t maneuver,
232 iii_drone::control::behavior_state_t behavior_state
233) const {
234
235 if (behavior_state.maneuver_is_executing) {
236
237 return false;
238
239 }
240
241 switch(maneuver.type) {
242
243 case maneuver::MANEUVER_TYPE_TAKEOFF: {
244
245 std::shared_ptr<takeoff_maneuver_params_t> takeoff_maneuver_params = std::static_pointer_cast<takeoff_maneuver_params_t>(maneuver.params);
246 float target_altitude = takeoff_maneuver_params->target_altitude;
247
248 bool target_altitude_valid = target_altitude >= parameters_->minimum_target_altitude();
249
250 return behavior_state.armed &&
251 behavior_state.offboard &&
252 behavior_state.on_ground() &&
253 target_altitude_valid;
254
255 }
256
257 case maneuver::MANEUVER_TYPE_LANDING:
258 return behavior_state.in_flight();
259
260 case maneuver::MANEUVER_TYPE_FLY_TO_POSITION: {
261
262 std::shared_ptr<fly_to_position_maneuver_params_t> fly_to_position_maneuver_params = std::static_pointer_cast<fly_to_position_maneuver_params_t>(maneuver.params);
263
264 point_t target_position_in_world_frame = fly_to_position_maneuver_params->transform_target_position(
265 parameters_->world_frame_id(),
266 tf_buffer_
267 );
268
269 bool target_position_valid = target_position_in_world_frame[2] - ground_altitude_estimate_->Load() >= parameters_->minimum_target_altitude();
270
271 return behavior_state.in_flight() &&
272 behavior_state.offboard &&
273 target_position_valid;
274
275 }
276
277 case maneuver::MANEUVER_TYPE_FLY_TO_OBJECT:
278 return behavior_state.in_flight() &&
279 behavior_state.offboard;
280
281 case maneuver::MANEUVER_TYPE_CABLE_LANDING:
282 return behavior_state.in_flight() &&
283 behavior_state.offboard &&
284 behavior_state.has_target_cable &&
285 behavior_state.target_cable_position_known;
286
287 case maneuver::MANEUVER_TYPE_CABLE_TAKEOFF:
288 return behavior_state.on_cable() &&
289 behavior_state.offboard &&
290 behavior_state.armed &&
291 behavior_state.has_target_cable &&
292 behavior_state.target_cable_position_known;
293
294 case maneuver::MANEUVER_TYPE_DISARM_ON_CABLE:
295 return behavior_state.on_cable() &&
296 behavior_state.offboard &&
297 behavior_state.armed &&
298 behavior_state.has_target_cable &&
299 behavior_state.target_cable_position_known;
300
301 case maneuver::MANEUVER_TYPE_ARM_ON_CABLE:
302 return behavior_state.on_cable() &&
303 !behavior_state.armed &&
304 behavior_state.has_target_cable &&
305 behavior_state.target_cable_position_known;
306
307 default:
308 return true;
309
310 }
