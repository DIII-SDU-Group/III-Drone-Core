39behavior_state_t BehaviorState::ProjectExpectedBehaviorState(maneuver_t maneuver) const {
40
41 std::vector<maneuver_t> maneuver_queue = (*maneuver_queue_).vector();
42
43 maneuver_t current_maneuver = current_maneuver_->Load();
44
45 if (maneuver_queue.size() == 0 || maneuver_queue[0] != current_maneuver) {
46
47 maneuver_queue.insert(
48 maneuver_queue.begin(),
49 current_maneuver
50 );
51
52 }
53
54 maneuver_queue.push_back(maneuver);
55
56 behavior_state_t bhst = behavior_state();
57 bhst.maneuver_is_executing = false;
58 bhst.current_maneuver = maneuver_t();
59
60 for (unsigned int i = 0; i < maneuver_queue.size(); i++) {
61
62 if (
63 !maneuverCanExecute(
64 maneuver_queue[i],
65 bhst
66 )
67 ) {
68
69 throw std::runtime_error("BehaviorState::ProjectExpectedBehaviorState: maneuver queue is invalid");
70
71 }
72
73 bhst = ProjectExpectedBehaviorState(
74 maneuver_queue[i],
75 bhst
76 );
77
78 }
79
80 return bhst;

84behavior_state_t BehaviorState::ProjectExpectedBehaviorState(
85 maneuver_t maneuver,
86 behavior_state_t bhst
87) const {
88
89 behavior_state_t bhst_out = bhst;
90
91 bhst_out.maneuver_is_executing = false;
92 bhst_out.current_maneuver = maneuver_t();
93
94 switch(maneuver.type) {
95
96 case maneuver::MANEUVER_TYPE_TAKEOFF:
97 bhst_out.armed = true;
98 bhst_out.offboard = true;
99 bhst_out.drone_location = DRONE_LOCATION_IN_FLIGHT;
100 break;
101
102 case maneuver::MANEUVER_TYPE_LANDING:
103 bhst_out.armed = false;
104 bhst_out.offboard = false;
105 bhst_out.drone_location = DRONE_LOCATION_ON_GROUND;
106 break;
107
108 case maneuver::MANEUVER_TYPE_FLY_TO_POSITION:
109 bhst_out.armed = true;
110 bhst_out.offboard = true;
111 bhst_out.drone_location = DRONE_LOCATION_IN_FLIGHT;
112 break;
113
114 case maneuver::MANEUVER_TYPE_FLY_TO_OBJECT:
115 bhst_out.armed = true;
116 bhst_out.offboard = true;
117 bhst_out.drone_location = DRONE_LOCATION_IN_FLIGHT;
118 break;
119
120 case maneuver::MANEUVER_TYPE_CABLE_LANDING:
121 bhst_out.armed = true;
122 bhst_out.offboard = true;
123 bhst_out.drone_location = DRONE_LOCATION_ON_CABLE;
124 break;

126 case maneuver::MANEUVER_TYPE_CABLE_TAKEOFF:
127 bhst_out.armed = true;
128 bhst_out.offboard = true;
129 bhst_out.drone_location = DRONE_LOCATION_IN_FLIGHT;
130 break;
131
132 case maneuver::MANEUVER_TYPE_DISARM_ON_CABLE:
133 bhst_out.armed = false;
134 bhst_out.offboard = false;
135 bhst_out.drone_location = DRONE_LOCATION_ON_CABLE;
136 break;

139 bhst_out.armed = true;
140 bhst_out.offboard = true;
141 bhst_out.drone_location = DRONE_LOCATION_ON_CABLE;

256int BehaviorState::on_cable_id() const {
257
258 if (vehicle_odometry_adapter_history_->empty()) {
259
260 return false;
261
262 }
263
264 if (powerline_adapter_history_->empty()) {
265
266 return false;
267
268 }
269
270 iii_drone::adapters::px4::VehicleOdometryAdapter vehicle_odometry_adapter = (*vehicle_odometry_adapter_history_)[0];
271
272 point_t drone_position = vehicle_odometry_adapter.position();
273
274 geometry_msgs::msg::TransformStamped transform_stamped = tf_buffer_->lookupTransform(
275 parameters_->cable_gripper_frame_id(),
276 parameters_->drone_frame_id(),
277 tf2::TimePointZero
278 );
279
280 vector_t v_drone_to_gripper = vectorFromTransformMsg(transform_stamped.transform);
281
282 point_t gripper_position = drone_position + v_drone_to_gripper;
283
284 iii_drone::adapters::PowerlineAdapter powerline_adapter = (*powerline_adapter_history_)[0];
285
286 iii_drone::adapters::SingleLineAdapter closest_line;
287
288 try {
289 closest_line = powerline_adapter.GetClosestLine(gripper_position);
290 } catch (std::runtime_error & e) {
291 return -1;
292 }
293
294 point_t closest_line_position = closest_line.position();
295
296 if ((closest_line_position - gripper_position).norm() <= parameters_->on_cable_max_euc_distance()) {
297
298 return closest_line.id();
299
300 }
301
302 return -1;
303

383bool BehaviorState::maneuverCanExecute(
384 iii_drone::control::maneuver::maneuver_t maneuver,
385 iii_drone::control::behavior_state_t behavior_state
386) const {
387
388 if (behavior_state.maneuver_is_executing) {
389
390 return false;
391
392 }
393
394 switch(maneuver.type) {
395
396 case maneuver::MANEUVER_TYPE_TAKEOFF: {
397
398 std::shared_ptr<takeoff_maneuver_params_t> takeoff_maneuver_params = std::static_pointer_cast<takeoff_maneuver_params_t>(maneuver.params);
399 float target_altitude = takeoff_maneuver_params->target_altitude;
400
401 bool target_altitude_valid = target_altitude >= parameters_->minimum_target_altitude();
402
403 return behavior_state.armed &&
404 behavior_state.offboard &&
405 behavior_state.on_ground() &&
406 target_altitude_valid;
407
408 }
409
410 case maneuver::MANEUVER_TYPE_LANDING:
411 return behavior_state.in_flight();
412
413 case maneuver::MANEUVER_TYPE_FLY_TO_POSITION: {
414
415 std::shared_ptr<fly_to_position_maneuver_params_t> fly_to_position_maneuver_params = std::static_pointer_cast<fly_to_position_maneuver_params_t>(maneuver.params);
416
417 point_t target_position_in_world_frame = fly_to_position_maneuver_params->transform_target_position(
418 parameters_->world_frame_id(),
419 tf_buffer_
420 );
421
422 bool target_position_valid = target_position_in_world_frame[2] - ground_altitude_estimate_->Load() >= parameters_->minimum_target_altitude();
423
424 return behavior_state.in_flight() &&
425 behavior_state.offboard &&
426 target_position_valid;
427
428 }
429
430 case maneuver::MANEUVER_TYPE_FLY_TO_OBJECT:
431 return behavior_state.in_flight() &&
432 behavior_state.offboard;
433
434 case maneuver::MANEUVER_TYPE_CABLE_LANDING:
435 return behavior_state.in_flight() &&
436 behavior_state.offboard &&
437 behavior_state.has_target_cable &&
438 behavior_state.target_cable_position_known;
439
440 case maneuver::MANEUVER_TYPE_CABLE_TAKEOFF:
441 return behavior_state.on_cable() &&
442 behavior_state.offboard &&
443 behavior_state.armed &&
444 behavior_state.has_target_cable &&
445 behavior_state.target_cable_position_known;
446
447 case maneuver::MANEUVER_TYPE_DISARM_ON_CABLE:
448 return behavior_state.on_cable() &&
449 behavior_state.offboard &&
450 behavior_state.armed &&
451 behavior_state.has_target_cable &&
452 behavior_state.target_cable_position_known;
453
454 case maneuver::MANEUVER_TYPE_ARM_ON_CABLE:
455 return behavior_state.on_cable() &&
456 !behavior_state.armed &&
457 behavior_state.has_target_cable &&
458 behavior_state.target_cable_position_known;
459
460 default:
461 return true;
462
463 }
