48 if constexpr (std::is_same<ActionT, iii_drone_interfaces::action::Takeoff>::value) {

150point_t fly_to_position_maneuver_params_t::transform_target_position(
151 const std::string target_frame_id,
152 tf2_ros::Buffer::SharedPtr tf_buffer
153) const {
154
155 if (frame_id == target_frame_id) {
156 return target_position;
157 }
158
159 geometry_msgs::msg::PointStamped target_position_stamped;
160 target_position_stamped.header.frame_id = frame_id;
161 target_position_stamped.point = pointMsgFromPoint(target_position);
162
163 geometry_msgs::msg::PointStamped transformed_target_position_stamped;
164
165 try {
166 transformed_target_position_stamped = tf_buffer->transform(target_position_stamped, target_frame_id);
167 } catch (tf2::TransformException & ex) {
168 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "%s", ex.what());
169 throw ex;
170 }
171
172 return pointFromPointMsg(transformed_target_position_stamped.point);
173
174}
