39point_t fly_to_position_maneuver_params_t::transform_target_position(
40 const std::string target_frame_id,
41 tf2_ros::Buffer::SharedPtr tf_buffer
42) const {
43
44 if (frame_id == target_frame_id) {
45 return target_position;
46 }
47
48 geometry_msgs::msg::PointStamped target_position_stamped;
49 target_position_stamped.header.frame_id = frame_id;
50 target_position_stamped.point = pointMsgFromPoint(target_position);
51
52 geometry_msgs::msg::PointStamped transformed_target_position_stamped;
53
54 try {
55 transformed_target_position_stamped = tf_buffer->transform(target_position_stamped, target_frame_id);
56 } catch (tf2::TransformException & ex) {
57 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "%s", ex.what());
58 throw ex;
59 }
60
61 return pointFromPointMsg(transformed_target_position_stamped.point);
