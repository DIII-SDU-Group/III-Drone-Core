303void ManeuverControllerNodeConfigurator::updateParametersObjectsCallback(const rclcpp::Parameter & parameter) {
304
305 std::string param_name = parameter.get_name();
306
307 if (param_name == "/control/maneuver_controller/landed_altitude_threshold") {
308
309 combined_drone_awareness_handler_parameters_->landed_altitude_threshold() = parameter.as_double();
310
311 } else if (param_name == "/control/maneuver_controller/on_cable_max_euc_distance") {
312
313 combined_drone_awareness_handler_parameters_->on_cable_max_euc_distance() = parameter.as_double();
314
315 } else if (param_name == "/control/maneuver_controller/hover_by_object_max_euc_dist") {
316
317 maneuver_scheduler_parameters_->hover_by_object_max_euc_dist() = parameter.as_double();
318
319 } else if (param_name == "/control/maneuver_controller/hover_by_object_max_euc_dist") {
320
321 maneuver_scheduler_parameters_->hover_by_object_max_euc_dist() = parameter.as_double();
322
323 } else if (param_name == "/control/maneuver_controller/hover_on_cable_default_z_velocity") {
324
325 maneuver_scheduler_parameters_->hover_on_cable_default_z_velocity() = parameter.as_double();
326
327 } else if (param_name == "/control/maneuver_controller/hover_on_cable_default_yaw_rate") {
328
329 maneuver_scheduler_parameters_->hover_on_cable_default_yaw_rate() = parameter.as_double();
330
331 } else if (param_name == "/control/maneuver_controller/reached_position_euclidean_distance_threshold") {
332
333 fly_to_position_maneuver_server_parameters_->reached_position_euclidean_distance_threshold() = parameter.as_double();
334 fly_to_object_maneuver_server_parameters_->reached_position_euclidean_distance_threshold() = parameter.as_double();
335
336 } else if (param_name == "/control/maneuver_controller/minimum_target_altitude") {
337
338 fly_to_position_maneuver_server_parameters_->minimum_target_altitude() = parameter.as_double();
339 fly_to_object_maneuver_server_parameters_->minimum_target_altitude() = parameter.as_double();
340
341 }
342
343 after_parameter_change_callback_(parameter);
