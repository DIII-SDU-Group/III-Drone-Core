318 def gripper_command_srv_callback(self, request: GripperCommand.Request, response: GripperCommand.Response):
319 self.get_logger().info("Received gripper command: " + str(request.gripper_command))
320
321 gripper_command = request.gripper_command

324 if (gripper_command == GripperCommand.Request.GRIPPER_COMMAND_OPEN):

339 start_time = self.get_clock().now()
