131iii_drone::types::transform_matrix_t CombinedDroneAwarenessHandler::ComputeTargetTransform(const iii_drone::adapters::TargetAdapter & target_adapter) const {
132
133 iii_drone::control::Reference reference;
134
135 if (target_adapter.target_type() == iii_drone::adapters::TARGET_TYPE_CABLE) {
136
137 iii_drone::adapters::PowerlineAdapter powerline_adapter = (*powerline_adapter_history_)[0];
138
139 iii_drone::adapters::SingleLineAdapter target_line;
140
141 try {
142
143 target_line = powerline_adapter.GetLine(target_adapter.target_id());
144
145 } catch (std::runtime_error & e) {
146
147 throw std::runtime_error("CombinedDroneAwarenessHandler::ComputeTargetReference: target line with id " + std::to_string(target_adapter.target_id()) + " not found");
148 }
149
150 geometry_msgs::msg::PoseStamped target_object_pose_stamped = target_line.ToPoseStampedMsg();
151
152 geometry_msgs::msg::PoseStamped target_object_pose_stamped_world;
153
154 try {
155
156 target_object_pose_stamped_world = tf_buffer_->transform(
157 target_object_pose_stamped,
158 params_->world_frame_id()
159 );
160
161 } catch (tf2::TransformException & e) {
162
163 throw std::runtime_error("CombinedDroneAwarenessHandler::ComputeTargetReference: could not transform target object pose to world frame: " + std::string(e.what()));
164
165 }
166
167 iii_drone::types::transform_matrix_t d_T_c = target_adapter.target_transform();
168 iii_drone::types::pose_t w_T_c_pose = poseFromPoseMsg(target_object_pose_stamped_world.pose);
169 iii_drone::types::transform_matrix_t w_T_c = createTransformMatrix(
170 w_T_c_pose.position,
171 w_T_c_pose.orientation
172 );
173 iii_drone::types::transform_matrix_t w_T_d = w_T_c * d_T_c.inverse();
174
175 return w_T_d;
176
177 } else {
178
179 throw std::runtime_error("CombinedDroneAwarenessHandler::ComputeTargetReference: target type not implemented");
180
181 }

411void CombinedDroneAwarenessHandler::updateGroundAltitudeEstimate(
412 bool armed,
413 bool offboard,
414 bool has_target_cable
415) {
416
417 bool is_on_ground = !armed && !offboard && !has_target_cable;
418
419 if (!is_on_ground) {
420 ground_altitude_update_timer_->cancel();
421 return;
422 }
423
424 if (vehicle_odometry_adapter_history_->empty()) {
425 return;
426 }
427
428 if (!ground_altitude_update_timer_->is_canceled()) {
429 return;
430 }
431
432 double ground_altitude = (*vehicle_odometry_adapter_history_)[0].position()[2];
433
434 ground_altitudes_history_.Store(ground_altitude);
435
436 std::vector<double> ground_altitudes = ground_altitudes_history_.vector();
437
438 double ground_altitude_estimate = std::accumulate(ground_altitudes.begin(), ground_altitudes.end(), 0.0) / ground_altitudes.size();
439
440 ground_altitude_estimate_->Store(ground_altitude_estimate);
441
442 ground_altitude_update_timer_->reset();
443
444 geometry_msgs::msg::TransformStamped ground_tf;
445
446 ground_tf.header.stamp = node_->now();
447 ground_tf.header.frame_id = params_->world_frame_id();
448 ground_tf.child_frame_id = params_->ground_frame_id();
449
450 ground_tf.transform = transformMsgFromTransform(
451 vector_t(0, 0, ground_altitude_estimate),
452 quaternion_t(1, 0, 0, 0)
453 );
454
455 tf_broadcaster_->sendTransform(ground_tf);
456
457}

459void CombinedDroneAwarenessHandler::updateDroneLocation(combined_drone_awareness_t & combined_drone_awareness) {
460
461 if (vehicle_odometry_adapter_history_->empty() || vehicle_status_adapter_history_->empty()) {
462 combined_drone_awareness.drone_location = DRONE_LOCATION_UNKNOWN;
463 combined_drone_awareness.on_cable_id = -1;
464 return;
465 }
466
467 iii_drone::adapters::px4::VehicleOdometryAdapter vehicle_odometry_adapter = (*vehicle_odometry_adapter_history_)[0];
468
469 point_t drone_position = vehicle_odometry_adapter.position();
470
471 // Check if on ground:
472 if (drone_position[2] < params_->landed_altitude_threshold()) {
473 combined_drone_awareness.drone_location = DRONE_LOCATION_ON_GROUND;
474 combined_drone_awareness.on_cable_id = -1;
475 return;
476 }
477
478 // Check if on cable:
479 if (!powerline_adapter_history_->empty()) {
480
481 geometry_msgs::msg::TransformStamped transform_stamped = tf_buffer_->lookupTransform(
482 params_->cable_gripper_frame_id(),
483 params_->drone_frame_id(),
484 tf2::TimePointZero
485 );
486
487 vector_t v_drone_to_gripper = vectorFromTransformMsg(transform_stamped.transform);
488
489 point_t gripper_position = drone_position + v_drone_to_gripper;
490
491 iii_drone::adapters::PowerlineAdapter powerline_adapter = (*powerline_adapter_history_)[0];
492
493 iii_drone::adapters::SingleLineAdapter closest_line;
494
495 bool found_closest_line = false;
496
497 try {
498 closest_line = powerline_adapter.GetClosestLine(gripper_position);
499 found_closest_line = true;
500 } catch (std::runtime_error & e) {
501 found_closest_line = false;
502 }
503
504 if (found_closest_line) {
505
506 point_t closest_line_position = closest_line.position();
507
508 if ((closest_line_position - gripper_position).norm() <= params_->on_cable_max_euc_distance()) {
509
510 combined_drone_awareness.drone_location = DRONE_LOCATION_ON_CABLE;
511 combined_drone_awareness.on_cable_id = closest_line.id();
512
513 return;
514
515 }
516 }
517 }
518
519 // Check if in flight:
520 if (combined_drone_awareness.armed) {
521 combined_drone_awareness.drone_location = DRONE_LOCATION_IN_FLIGHT;
522 return;
523 }
524
525 // Throw error if no location found
526 std::string error_message = "CombinedDroneAwarenessHandler::updateDroneLocation(): Could not determine drone location.";
527
528 if (params_->fail_on_unable_to_locate()) {
529 RCLCPP_FATAL(node_->get_logger(), error_message.c_str());
530 throw std::runtime_error(error_message);
531 } else {
532 RCLCPP_ERROR(node_->get_logger(), error_message.c_str());
533 combined_drone_awareness.drone_location = DRONE_LOCATION_UNKNOWN;
534 combined_drone_awareness.on_cable_id = -1;
535 }
536
