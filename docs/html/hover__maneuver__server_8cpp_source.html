34bool HoverManeuverServer::CanExecuteManeuver(
35 const Maneuver & maneuver,
36 const combined_drone_awareness_t & drone_awareness
37) const {
38
39 if (maneuver.maneuver_type() != MANEUVER_TYPE_HOVER) {
40 return false;
41 }
42
43 if (!drone_awareness.offboard) {
44 return false;
45 }
46
47 if (!drone_awareness.armed) {
48 return false;
49 }
50
51 if (!drone_awareness.in_flight()) {
52 return false;
53 }
54
55 return true;
56
57}

185void HoverManeuverServer::publishResultAndFinalize(
186 Maneuver & maneuver,
187 maneuver_result_type_t maneuver_result_type
188) {
189
190 auto result = std::make_shared<Hover::Result>();
191 auto goal_handle = std::static_pointer_cast<rclcpp_action::ServerGoalHandle<Hover>>(maneuver.goal_handle());
192
193 switch (maneuver_result_type) {
194 case MANEUVER_RESULT_TYPE_SUCCEED:
195 goal_handle->succeed(result);
196 break;
197 case MANEUVER_RESULT_TYPE_ABORT:
198 goal_handle->abort(result);
199 break;
200 case MANEUVER_RESULT_TYPE_CANCEL:
201 goal_handle->canceled(result);
202 break;
203 }
204
205}
