150void PowerlineMapperNode::mmWaveCallback(const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
151
152 iii_drone::adapters::PointCloudAdapter pcl_adapter(msg);
153
154 std::vector<point_t> transformed_points;
155 std::vector<point_t> projected_points;
156
157 std::vector<point_t> pcl_points = pcl_adapter.points();
158
159 for (size_t i = 0; i < pcl_points.size(); i++) {
160
161 auto line = SingleLine(
162 -1,
163 pcl_points[i],
164 pl_direction_,
165 configurator_.mmwave_frame_id(),
166 tf_buffer_,
167 configurator_.powerline_parameters()
168 );
169
170 if(!line.IsInFOVStrict()) {
171
172 continue;
173
174 }
175
176 geometry_msgs::msg::PointStamped pt;
177 pt.header.frame_id = msg->header.frame_id;
178 pt.point = pointMsgFromPoint(pcl_points[i]);
179
180 pt = tf_buffer_->transform(
181 pt,
182 configurator_.drone_frame_id()
183 );
184
185 point_t transformed_point = pointFromPointMsg(pt.point);
186
187 point_t projected_point = powerline_.UpdateLine(transformed_point);
188
189 transformed_points.push_back(transformed_point);
190 projected_points.push_back(projected_point);
191
192 }
193
194 publishPoints(
195 transformed_points,
196 transformed_points_pub_
197 );
198
199 publishPoints(
200 projected_points,
201 projected_points_pub_
202 );
203
204 powerline_.CleanupLines();
205
206 powerline_.ComputeInterLinePositions();
207
208}
