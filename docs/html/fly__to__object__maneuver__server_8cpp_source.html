38bool FlyToObjectManeuverServer::CanExecuteManeuver(
39 const Maneuver & maneuver,
40 const combined_drone_awareness_t & drone_awareness
41) const {
42
43 if (maneuver.maneuver_type() != MANEUVER_TYPE_FLY_TO_OBJECT) {
44 return false;
45 }
46
47 auto cda_handler = awareness_handler();
48
49 fly_to_object_maneuver_params_t params(maneuver.maneuver_params());
50
51 if (params.target_adapter.target_type() != TARGET_TYPE_CABLE) {
52 return false;
53 }
54
55 if (!drone_awareness.offboard) {
56 return false;
57 }
58
59 if (!drone_awareness.armed) {
60 return false;
61 }
62
63 if (!drone_awareness.in_flight()) {
64 return false;
65 }
66
67 transform_matrix_t target_transform;
68
69 try {
70
71 target_transform = awareness_handler()->ComputeTargetTransform(target_adapter_);
72
73 } catch (const std::runtime_error &e) {
74
75 return false;
76
77 }
78
79 point_t target_position_in_world_frame = target_transform.block<3, 1>(0, 3);
80
81 bool target_position_valid = target_position_in_world_frame[2] - cda_handler->ground_altitude_estimate() >= parameters_->minimum_target_altitude();
82
83 return target_position_valid;
84
85}

142Reference FlyToObjectManeuverServer::computeReference(const State & state) {
143
144 Reference target_reference = getUpdatedTargetReference(state);
145
146 Reference ref;
147
148 if (parameters_->generate_trajectories_asynchronously_with_delay()) {
149
150 if (first_iteration_) {
151
152 if (trajectory_generator_client_->busy()) {
153
154 std::string error_message = "FlyToObjectManeuverServer::computeReference(): Trajectory generator client is busy on first iteration, cannot start execution of maneuver.";
155
156 RCLCPP_FATAL(node()->get_logger(), error_message.c_str());
157
158 throw std::runtime_error(error_message);
159
160 }
161
162 trajectory_generator_client_->Reset(state);
163
164 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[0];
165
166 trajectory_generator_client_->ComputeReferenceTrajectoryAsync(
167 state,
168 target_reference,
169 true,
170 true,
171 MPC_mode_t::positional
172 );
173
174 first_iteration_ = false;
175
176 } else {
177
178 while (!trajectory_generator_client_->done()) {
179
180 rclcpp::sleep_for(std::chrono::milliseconds(parameters_->generate_trajectories_poll_period_ms()));
181
182 }
183
184 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[1];
185
186 trajectory_generator_client_->ComputeReferenceTrajectoryAsync(
187 state,
188 target_reference,
189 true,
190 false,
191 MPC_mode_t::positional
192 );
193
194 }
195
196 } else {
197
198 if (trajectory_generator_client_->busy()) {
199
200 std::string error_message = "FlyToObjectManeuverServer::computeReference(): Trajectory generator client is busy on first iteration, cannot start execution of maneuver.";
201
202 RCLCPP_FATAL(node()->get_logger(), error_message.c_str());
203
204 throw std::runtime_error(error_message);
205
206 }
207
208 bool first_it = first_iteration_;
209
210 if (first_it) {
211
212 trajectory_generator_client_->Reset(state);
213
214 first_iteration_ = false;
215
216 }
217
218 trajectory_generator_client_->ComputeReferenceTrajectoryBlocking(
219 state,
220 target_reference,
221 true,
222 first_it,
223 MPC_mode_t::positional,
224 parameters_->generate_trajectories_poll_period_ms()
225 );
226
227 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[0];
228
229 }
230
231 return ref;
232
233}

235bool FlyToObjectManeuverServer::hasSucceeded(Maneuver & maneuver) {
236
237 auto cda_handler = awareness_handler();
238
239 State state = cda_handler->GetState();
240
241 Reference target_reference = getUpdatedTargetReference(state);
242
243 if (hasFailed(maneuver)) {
244 return false;
245 }
246
247 Eigen::Vector4d euc_pos = {
248 state.position()[0],
249 state.position()[1],
250 state.position()[2],
251 state.yaw()
252 };
253
254 Eigen::Vector4d target_euc_pos = {
255 target_reference.position()[0],
256 target_reference.position()[1],
257 target_reference.position()[2],
258 target_reference.yaw()
259 };
260
261 double distance = (euc_pos - target_euc_pos).norm();
262
263 return distance < parameters_->reached_position_euclidean_distance_threshold();
264
265}

280void FlyToObjectManeuverServer::publishFeedback(Maneuver & maneuver) {
281
282 auto feedback = std::make_shared<iii_drone_interfaces::action::FlyToObject::Feedback>();
283
284 ReferenceTrajectory reference_trajectory = trajectory_generator_client_->GetReferenceTrajectory();
285 ReferenceTrajectoryAdapter reference_trajectory_adapter(reference_trajectory);
286
287 State state = awareness_handler()->GetState();
288 Reference target_reference = getUpdatedTargetReference(state);
289
290 feedback->planned_path = reference_trajectory_adapter.ToPathMsg(parameters_->world_frame_id());
291 feedback->vehicle_pose = StateAdapter(awareness_handler()->GetState()).ToPoseStampedMsg(parameters_->world_frame_id());
292 feedback->distance_vehicle_to_target = (target_reference.position() - state.position()).norm();
293
294 auto goal_handle = std::static_pointer_cast<GoalHandleFlyToObject>(maneuver.goal_handle());
295
296 goal_handle->publish_feedback(feedback);

300void FlyToObjectManeuverServer::publishResultAndFinalize(
301 Maneuver & maneuver,
302 maneuver_result_type_t maneuver_result_type
303) {
304
305 auto result = std::make_shared<iii_drone_interfaces::action::FlyToObject::Result>();
306 auto goal_handle = std::static_pointer_cast<GoalHandleFlyToObject>(maneuver.goal_handle());
307
308 switch (maneuver_result_type) {
309 case MANEUVER_RESULT_TYPE_SUCCEED:
310 result->success = true;
311 goal_handle->succeed(result);
312 break;
313 case MANEUVER_RESULT_TYPE_ABORT:
314 result->success = false;
315 goal_handle->abort(result);
316 awareness_handler()->ClearTarget();
317 break;
318 case MANEUVER_RESULT_TYPE_CANCEL:
319 result->success = false;
320 goal_handle->canceled(result);
321 awareness_handler()->ClearTarget();
322 break;
323 }
324

345Reference FlyToObjectManeuverServer::getUpdatedTargetReference(const iii_drone::control::State & state) {
346
347 auto cda_handler = awareness_handler();
348
349 transform_matrix_t target_transform;
350
351 try {
352
353 target_transform = cda_handler->ComputeTargetTransform(target_adapter_);
354
355 } catch (const std::runtime_error &e) {
356
357 has_failed_ = true;
358
359 return Reference(
360 state.position(),
361 state.yaw()
362 );
363
364 }
365
366 point_t target_position_in_world_frame = target_transform.block<3, 1>(0, 3);
367 quaternion_t target_orientation_in_world_frame = quaternionFromTransformMatrix(target_transform);
368 double target_yaw = quatToEul(target_orientation_in_world_frame)[2];
369
370 return Reference(
371 target_position_in_world_frame,
372 target_yaw
373 );
