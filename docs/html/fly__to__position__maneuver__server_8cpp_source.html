37bool FlyToPositionManeuverServer::CanExecuteManeuver(
38 const Maneuver & maneuver,
39 const combined_drone_awareness_t & drone_awareness
40) const {
41
42 if (maneuver.maneuver_type() != MANEUVER_TYPE_FLY_TO_POSITION) {
43 return false;
44 }
45
46 auto cda_handler = awareness_handler();
47
48 std::shared_ptr<fly_to_position_maneuver_params_t> fly_to_position_maneuver_params = std::static_pointer_cast<fly_to_position_maneuver_params_t>(maneuver.maneuver_params());
49
50 point_t target_position_in_world_frame = fly_to_position_maneuver_params->transform_target_position(
51 parameters_->world_frame_id(),
52 cda_handler->tf_buffer()
53 );
54
55 bool target_position_valid = target_position_in_world_frame[2] - cda_handler->ground_altitude_estimate() >= parameters_->minimum_target_altitude();
56
57 return drone_awareness.in_flight() &&
58 drone_awareness.offboard &&
59 target_position_valid;
60
61}

117Reference FlyToPositionManeuverServer::computeReference(const State & state) {
118
119 Reference ref;
120
121 if (parameters_->generate_trajectories_asynchronously_with_delay()) {
122
123 if (first_iteration_) {
124
125 if (trajectory_generator_client_->busy()) {
126
127 std::string error_message = "FlyToPositionManeuverServer::computeReference(): Trajectory generator client is busy on first iteration, cannot start execution of maneuver.";
128
129 RCLCPP_FATAL(node()->get_logger(), error_message.c_str());
130
131 throw std::runtime_error(error_message);
132
133 }
134
135 trajectory_generator_client_->Reset(state);
136
137 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[0];
138
139 trajectory_generator_client_->ComputeReferenceTrajectoryAsync(
140 state,
141 target_reference_,
142 true,
143 true,
144 MPC_mode_t::positional
145 );
146
147 first_iteration_ = false;
148
149 } else {
150
151 while (!trajectory_generator_client_->done()) {
152
153 rclcpp::sleep_for(std::chrono::milliseconds(parameters_->generate_trajectories_poll_period_ms()));
154
155 }
156
157 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[1];
158
159 trajectory_generator_client_->ComputeReferenceTrajectoryAsync(
160 state,
161 target_reference_,
162 false,
163 false,
164 MPC_mode_t::positional
165 );
166
167 }
168
169 } else {
170
171 if (trajectory_generator_client_->busy()) {
172
173 std::string error_message = "FlyToPositionManeuverServer::computeReference(): Trajectory generator client is busy on first iteration, cannot start execution of maneuver.";
174
175 RCLCPP_FATAL(node()->get_logger(), error_message.c_str());
176
177 throw std::runtime_error(error_message);
178
179 }
180
181 bool first_it = first_iteration_;
182
183 if (first_it) {
184
185 trajectory_generator_client_->Reset(state);
186
187 first_iteration_ = false;
188
189 }
190
191 trajectory_generator_client_->ComputeReferenceTrajectoryBlocking(
192 state,
193 target_reference_,
194 first_it,
195 first_it,
196 MPC_mode_t::positional,
197 parameters_->generate_trajectories_poll_period_ms()
198 );
199
200 ref = trajectory_generator_client_->GetReferenceTrajectory().references()[0];
201
202 }
203
204 return ref;

208bool FlyToPositionManeuverServer::hasSucceeded(Maneuver &) {
209
210 auto cda_handler = awareness_handler();
211
212 State state = cda_handler->GetState();
213
214 Eigen::Vector4d euc_pos = {
215 state.position()[0],
216 state.position()[1],
217 state.position()[2],
218 state.yaw()
219 };
220
221 Eigen::Vector4d target_euc_pos = {
222 target_reference_->position()[0],
223 target_reference_->position()[1],
224 target_reference_->position()[2],
225 target_reference_->yaw()
226 };
227
228 double distance = (euc_pos - target_euc_pos).norm();
229
230 return distance < parameters_->reached_position_euclidean_distance_threshold();
231
232}

261void FlyToPositionManeuverServer::publishResultAndFinalize(
262 Maneuver & maneuver,
263 maneuver_result_type_t maneuver_result_type
264) {
265
266 auto result = std::make_shared<iii_drone_interfaces::action::FlyToPosition::Result>();
267 auto goal_handle = std::static_pointer_cast<GoalHandleFlyToPosition>(maneuver.goal_handle());
268
269 switch (maneuver_result_type) {
270 case MANEUVER_RESULT_TYPE_SUCCEED:
271 result->success = true;
272 goal_handle->succeed(result);
273 break;
274 case MANEUVER_RESULT_TYPE_ABORT:
275 result->success = false;
276 goal_handle->abort(result);
277 break;
278 case MANEUVER_RESULT_TYPE_CANCEL:
279 result->success = false;
280 goal_handle->canceled(result);
281 break;
282 }
