342ReferenceTrajectory TrajectoryGenerator::postProcessMPC(
343 rclcpp::Time t0,
344 int N,
345 double dt,
346 double *u,
347 double *planned_traj,
348 double target_yaw
349) {
350
351 std::vector<Reference> trajectory_vector;
352
353 trajectory_vector.resize(N);
354
355 rclcpp::Time stamp = t0;
356
357 for (int i = 0; i < N; i++) {
358
359 point_t reference_position(
360 planned_traj[i*6+0],
361 planned_traj[i*6+1],
362 planned_traj[i*6+2]
363 );
364
365 vector_t reference_velocity(
366 planned_traj[i*6+3],
367 planned_traj[i*6+4],
368 planned_traj[i*6+5]
369 );
370
371 Reference ref;
372
373 if (i == 0) {
374
375 vector_t reference_acceleration(
376 u[0],
377 u[1],
378 u[2]
379 );
380
381 ref = Reference(
382 reference_position,
383 target_yaw,
384 reference_velocity,
385 0,
386 reference_acceleration,
387 NAN,
388 stamp
389 );
390
391 } else {
392
393 ref = Reference(
394 reference_position,
395 target_yaw,
396 reference_velocity,
397 0,
398 vector_t::Zero(),
399 NAN,
400 stamp
401 );
402
403 }
404
405 trajectory_vector.push_back(ref);
406
407 stamp = rclcpp::Time(stamp.nanoseconds() + dt*1e9);
408
409
410 }
411
412 return ReferenceTrajectory(trajectory_vector);
